function createGitHubIssue(username, pickedNumber) { const repoOwner = "ahraming"; // 본인의 GitHub 사용자명 const repoName = "-"; // 저장소 이름 fetch(`https://api.github.com/repos/${repoOwner}/${repoName}/issues`, { method: "POST", headers: { "Authorization": `token ${process.env.GITHUB_TOKEN}`, // GitHub Secrets에서 가져오기 "Accept": "application/vnd.github.v3+json", "Content-Type": "application/json" }, body: JSON.stringify({ title: `발표 순서 배정: ${username}`, body: `${username} 님이 발표 순서 ${pickedNumber}을(를) 배정받았습니다.` }) }) .then(response => response.json()) .then(data => console.log("Issue Created:", data)) .catch(error => console.error("Error creating issue:", error)); }
